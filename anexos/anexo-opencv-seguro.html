Configuración Segura de OpenCV

Introducción

OpenCV (Open Source Computer Vision Library) es una de las librerías más utilizadas para visión por computadora. Sin embargo, su configuración por defecto puede presentar vulnerabilidades de seguridad si no se configura adecuadamente.

⚠️ Importancia de la Seguridad en OpenCV

OpenCV interactúa directamente con: - Cámaras y dispositivos de entrada - Archivos de imagen y video - Redes (en configuraciones cliente/servidor) - Memoria del sistema

Una configuración insegura puede permitir: - Acceso no autorizado a cámaras - Ejecución de código arbitrario - Exposición de datos sensibles - Escalamiento de privilegios

Configuración Segura

1. Permisos de Archivo

# Restringir permisos de archivos de OpenCV
chmod 700 /usr/local/lib/python3.*/site-packages/cv2/
chmod 600 /usr/local/lib/python3.*/site-packages/cv2/*.so

# Proteger archivos de configuración
chmod 600 /etc/opencv.conf

2. Variables de Entorno Seguras

# Desactivar funcionalidades innecesarias
export OPENCV_VIDEOIO_DEBUG=0
export OPENCV_LOG_LEVEL=ERROR
export OPENCV_ENABLE_NONFREE=0

# Limitar memoria
export OPENCV_MAX_THREADS=4

3. Configuración de Python Seguro

import cv2
import os

# Desactivar funciones peligrosas
cv2.setNumThreads(4)  # Limitar hilos

# Configurar logging seguro
cv2.utils.switchToVerbose(False)

# Validar entrada antes de procesar
def secure_image_read(filepath):
    """Lectura segura de imágenes con validación"""
    # Verificar extensión
    allowed_extensions = ['.jpg', '.jpeg', '.png', '.bmp', '.tiff']
    ext = os.path.splitext(filepath)[1].lower()
    
    if ext not in allowed_extensions:
        raise ValueError(f"Extensión no permitida: {ext}")
    
    # Verificar tamaño máximo (10MB)
    if os.path.getsize(filepath) > 10 * 1024 * 1024:
        raise ValueError("Archivo demasiado grande")
    
    # Verificar magic bytes
    with open(filepath, 'rb') as f:
        magic = f.read(4)
        
    valid_magic = {
        b'\xff\xd8\xff': 'jpeg',
        b'\x89PNG': 'png',
        b'BM': 'bmp',
        b'II\x2a\x00': 'tiff',
        b'MM\x00\x2a': 'tiff'
    }
    
    if magic[:4] not in valid_magic:
        raise ValueError("Tipo de archivo no válido")
    
    return cv2.imread(filepath)

4. Configuración de Cámaras Seguras

# Configuración segura de cámara
def secure_camera_config(camera_index=0):
    """Configuración segura para captura de video"""
    cap = cv2.VideoCapture(camera_index)
    
    if not cap.isOpened():
        raise RuntimeError("No se puede abrir la cámara")
    
    # Establecer límites seguros
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1920)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 1080)
    cap.set(cv2.CAP_PROP_FPS, 30)
    
    # Verificar configuración
    width = cap.get(cv2.CAP_PROP_FRAME_WIDTH)
    height = cap.get(cv2.CAP_PROP_FRAME_HEIGHT)
    
    # No permitir más de 4K
    if width * height > 3840 * 2160:
        cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1920)
        cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 1080)
    
    return cap

5. Hardening de OpenCV en Producción

# production_settings.py
import cv2
import os

class SecureOpenCV:
    """Wrapper para operaciones seguras con OpenCV"""
    
    def __init__(self):
        self._configure_security()
    
    def _configure_security(self):
        # Desactivar optimización assembly potencialmente insecure
        cv2.setUseOptimized(False)
        
        # Limitar tablas de lookup
        cv2.setNumThreads(os.cpu_count() or 4)
        
        # Desactivar fallbacks peligrosos
        os.environ['OPENCV_ALLOW_SSE2'] = '0' if os.environ.get('SECURE_MODE') else '1'
    
    def read_image(self, path, max_size_mb=10):
        """Lectura segura de imagen"""
        if os.path.getsize(path) > max_size_mb * 1024 * 1024:
            raise ValueError(f"Imagen excede {max_size_mb}MB")
        
        img = cv2.imread(path)
        if img is None:
            raise ValueError(f"No se pudo leer: {path}")
        
        # Verificar dimensiones
        if img.size > 50 * 1024 * 1024:  # 50MB en memoria
            raise ValueError("Imagen demasiado grande para procesar")
        
        return img
    
    def process_video(self, path):
        """Procesamiento seguro de video"""
        cap = cv2.VideoCapture(path)
        
        # Validar
        if not cap.isOpened():
            raise ValueError(f"No se pudo abrir: {path}")
        
        # Contador de frames para evitar loops infinitos
        frame_count = 0
        max_frames = 10000
        
        while frame_count < max_frames:
            ret, frame = cap.read()
            if not ret:
                break
            # Procesar frame
            frame_count += 1
        
        cap.release()
        return frame_count
